5    /**
6     * Represents the move log

9    public class ChessMoveLog {
10   
11       /**
12        * The move log
13        */
14       private ArrayList<ChessMove> log = new ArrayList<>();
15   
16       /**
17        * Used to create a new instance
18        */
19       public ChessMoveLog() {}
20   
21       /**
22        * Used to clone an already existing log
23        * @param log Log to clone
24        */
25       public ChessMoveLog(ChessMoveLog log) {

27       }
28   
29       /**
30        * Used to add a move to the log
31        * @param move Move to add to the log

33       public void add(ChessMove move) {

37       /**
38        * Used to get the list of moves that have happened
39        * @return Log as a string
40        */
41       public String asString() {
42           StringBuilder builder = new StringBuilder();
43   
44           builder.append("[");
45   
46           for (ChessMove move : this.log) {
47               builder.append(move.asString());
48               builder.append(", ");
49           }
50   
51           builder.append("]");
52   
53           return builder.toString();
54       }
55   
56       /**
57        * Used to get the last log
58        * @return The chess move
59        */
60       public ChessMove getLast() {
61           if (this.log.size() - 1 == -1) return null;
62           return this.log.get(this.log.size() - 1);
63       }
64   }
