1    package com.github.smuddgge.items.chessboard;
2    
3    import com.github.smuddgge.game.ChessBoard;
4    import com.github.smuddgge.game.ChessBoardTile;
5    import com.github.smuddgge.items.Item;
6    import com.github.smuddgge.positions.ModularPosition;
7    
8    import javax.swing.*;
9    import java.awt.*;
10   
11   /**
12    * Represents the interface for the chessboard
13    */
14   public abstract class ChessBoardItem extends Item {
15   
16       /**
17        * The chessboard panel
18        */
19       private JButton panel = new JButton();
20   
21       /**
22        * Used to create a new chessboard interface
23        * @param modularPosition The position of the chessboard
24        */
25       public ChessBoardItem(ModularPosition modularPosition) {
26           super(modularPosition);
27       }
28   
29       /**
30        * Creates the chessboard based on its position
31        */
32       private void create() {
33           this.panel = new JButton();
34   
35           this.panel.setBounds(
36                   this.getItemRegion().getMin().getX(),
37                   this.getItemRegion().getMin().getY(),
38                   this.modularPosition.getWidth(), this.modularPosition.getHeight()
39           );
40   
41           this.panel.setBackground(Color.white);
42           this.getBoard().render(this.panel);
43   
44           // Adding a listener to all buttons
45           for (Component component : this.panel.getComponents()) {
46               if (!(component instanceof JButton button)) continue;
47               button.addActionListener(event -> this.onClick(this.getBoard().getTileAt(this.getMouseLocation())));
48           }
49       }
50   
51       /**
52        * Used to get the board
53        */
54       public abstract ChessBoard getBoard();
55   
56       /**
57        * When the mouse is hovering over a tile
58        */
59       public abstract void onTileHover(ChessBoardTile tile);
60   
61       /**
62        * When the mouse clicks a tile
63        */
64       public abstract void onClick(ChessBoardTile tile);
65   
66       /**
67        * When the application ticks
68        */
69       public abstract void tick();
70   
71       /**
72        * Item hover events
73        */
74       @Override public void onHover() {}
75       @Override public void onNotHover() {}
76   
77       @Override
78       public void updateItem() {
79           this.tick();
80           if (this.getBoard() == null) return;
81           this.onTileHover(this.getBoard().getTileAt(this.getMouseLocation()));
82       }
83   
84       @Override
85       public void render(JPanel frame) {
86           this.create();
87           frame.add(this.panel);
88       }
89   
90       @Override
91       public Component getComponent() {
92           return this.panel;
93       }
94   }
